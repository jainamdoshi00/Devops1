l1 = list(map(int,input().split()))
l2 = list(map(int,input().split()))
w = l1[::-1]
v = l2[::-1]
m=""
q=""
for i in w:
    m = m + str(i)
for i in v:
    q=q + str(i)
p = int(q)    
t = int(m)
su = p + t
lt = str(su)
f=[]
for i in lt:
    wq = int(i)
    f.append(wq)
print(f[::-1])
